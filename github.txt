# comandos de git

# crear un repositorio
git init

# crear un archivo
touch archivo.txt

# agregar un archivo a un repositorio
git add archivo.txt

# commitear los cambios
git commit -m "mensaje"

# crear un repositorio remoto
git remote add origin https://github.com/nicolas-goutay/python-para-principiantes.git

# push a un repositorio remoto
git push -u origin master

# pull a un repositorio remoto
git pull origin master

# crear un repositorio local
git clone https://github.com/nicolas-goutay/python-para-principiantes.git

# crear un repositorio local a partir de un repositorio remoto
git clone https://github.com/nicolas-goutay/python-para-principiantes.git
